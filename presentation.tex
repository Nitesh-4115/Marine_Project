\begin{frame}{Project Overview \& Scope}
    \textbf{Objective:} Replicate results from \textit{Mathematics 2022, 10, 3332}.
    \vspace{0.5cm}
    \begin{itemize}
        \item Develop a robust 4-DOF mathematical hydrodynamic AUV model.
        \item Implement control strategies from scratch:
        \begin{itemize}
            \item RBFPDSMC (Single-loop PD Sliding Mode Control)
            \item RBFPIDSMC (Single-loop PID Sliding Mode Control)
            \item DLNNSMC (Double-Loop Neural Network SMC)
        \end{itemize}
        \item Evaluate trajectory tracking accuracy under realistic environmental noise and parametric uncertainties.
    \end{itemize}
\end{frame}

\begin{frame}{Mathematical Modeling}
    \textbf{4-DOF Non-linear Equations of Motion:}
    \begin{block}{Kinematics and Dynamics}
        \begin{equation}
            \dot{\eta} = R(\psi) v
        \end{equation}
        \begin{equation}
            M \dot{v} + C(v)v + D(v)v = \tau + \tau_{dist}
        \end{equation}
    \end{block}
    \vspace{0.3cm}
    \begin{itemize}
        \item $\eta = [x, y, z, \psi]^T$: Earth-fixed positions
        \item $v = [u, v_s, w, r]^T$: Body-fixed velocities
        \item $M, C(v), D(v)$: Inertia, Coriolis, and Hydrodynamic Damping
        \item $\tau_{dist}$: Unknown environmental disturbances
    \end{itemize}
\end{frame}

\begin{frame}{Proposed Control Strategy: DLNNSMC}
    \begin{block}{Double-Loop Architecture}
        \textbf{Outer Loop (Kinematic):}
        $v_d = R^{-1}(\psi) \left( \dot{\eta}_d + K_{p} e_\eta + K_{i} \int e_\eta dt \right)$ \\
        \vspace{0.2cm}
        \textbf{Inner Loop (Dynamic):}
        $s = e_v + C_{in} \int e_v dt$
    \end{block}
    \vspace{0.3cm}
    \begin{itemize}
        \item \textbf{Sliding Mode Control (SMC):} Provides strong robustness. We utilize a $\tanh(s)$ boundary layer to practically eliminate chattering.
        \item \textbf{RBF Neural Network:} Dynamically learns the unmodeled hydrodynamics and disturbances via the adaptation law $\dot{\hat{W}} = \Gamma h(v) s^T$.
    \end{itemize}
\end{frame}

\begin{frame}{Simulation Realism: Noise \& Uncertainty}
    To rigorously test robustness, we avoided "perfect simulation" pitfalls:
    \begin{itemize}
        \item \textbf{Model Mismatch:} Controllers operate with 15\% parameter error ($\hat{M} = 0.85 M$).
        \item \textbf{Sensor Noise:} Gaussian noise injected into position and velocity feedback to mimic real IMU/DVL sensors.
        \item \textbf{Ocean Currents:} Low-frequency harmonic wave drift applied continuously.
        \item \textbf{Actuator Saturation:} Torques clamped to realistic physical bounds.
    \end{itemize}
\end{frame}
